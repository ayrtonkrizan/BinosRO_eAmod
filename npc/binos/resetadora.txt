-	script	Resetador	-1,{
	if(isPremium() <1){
		set .@ResetStat, 75000;	// Zeny requeridos para resetar atributos.
		set .@ResetSkill,75000;	// Zeny requeridos para resetar habilidades.
		set .@ResetBoth, 100000;// Zeny requeridos para resetar ambos.
	}else{
		set .@ResetStat, 7000;	// Zeny requeridos para resetar atributos.
		set .@ResetSkill, 7000;	// Zeny requeridos para resetar habilidades.
		set .@ResetBoth, 10000;	// Zeny requeridos para resetar ambos.
	}
	mes "[Resetador]";
	mes "Olá "+strcharinfo(0)+", eu sou a Resetador.";
	mes "Resetar Atributos: "+ .@ResetStat +"z";
	mes "Resetar Habilidades: "+ .@ResetSkill +"z";
	mes "Resetar Ambos: "+ .@ResetBoth +"z";
	mes "Por favor, selecione o serviço que deseja:";
	next;
	switch(select("^FF3355Resetar Habilidades:Resetar Atributos:Resetar Ambos^000000:Cancelar")) {
	case 1:
		mes "[Resetador]";
		if (Zeny < .@ResetSkill) {
			mes "Desculpe-me, mas você não possui zeny suficiente para esta tarefa.";
			close;
		}
		set Zeny,Zeny-.@ResetSkill;
		sc_end SC_ALL;
		resetskill;
		mes "Pronto Atributos resetados, volte quando precisar de algum serviço.";
		close;
	case 2:
		mes "[Resetador]";
		if (Zeny < .@ResetStat) {
			mes "Desculpe-me, mas você não possui zeny suficiente para esta tarefa.";
			close;
		}
		set Zeny,Zeny-.@ResetStat;
		resetstatus;
		mes "Pronto Habilidades resetadas, volte quando precisar de algum serviço.";
		close;
	case 3:
		mes "[Resetador]";
		if (Zeny < .@ResetBoth) {
			mes "Desculpe-me, mas você não possui zeny suficiente para esta tarefa.";
			close;
		}
		set Zeny,Zeny-.@ResetBoth;
		sc_end SC_ALL;
		resetskill;
		resetstatus;
		mes "Pronto seus Atributos e suas Habilidades foram resetadas, volte quando precisar de algum serviço.";
		close;
	case 4:
		mes "[Resetador]";
		mes "Tudo bem, volte quando precisar de algum serviço.";
		close;
	}
}
