		mes "Certo, irei te teleportar para lá. Boa Sorte!";

		callsub RoundDone,"3º";
		end;

		callsub RoundDone,"4º";
		end;

	OnTimer420000:
		callsub RoundDone,"Último";
		end;

		callsub EventEnd;
		end;

		if ( @map$ == "job3_rune02" ) {
			recovery;
			set .@spoint$, getsavepoint(0);
			set .@x, getsavepoint(1);
			set .@y, getsavepoint(2);
			warp .@spoint$, .@x, .@y;
			end;
		}
		else{ 
			end; 
		}
		end;

		initnpctimer;
		announce "[Segurança do Calabouço] Quem for corajoso tem 5 minutos para entrar no calabouço!!!",bc_all,0xADD8E6;
		set $open,1;
		end;

		disablenpc "Mecânico";
		set $open,0;
		announce "[Evento Mecânico Pistola] O 1º Round Começou !",bc_all,0xADD8E6;

		end;
	
	EventEnd:
		set $open,0;
		killmonsterall "job3_rune02",0;
		switch(getmapusers("job3_rune02")){
			case 0:
				announce "[Segurança do Calabouço] Ninguém sobreviveu, boa sorte na próxima!", bc_all, 0xADD8E6;
				break;
			case 1:
				announce "[Segurança do Calabouço] Parabéns ao vencedor!!", bc_all, 0xADD8E6;
				monster "job3_rune02",39,40,"Baú do Tesouro",1798,1,"Treasure::OnPriceBoxBreak";
				break;
			default:
				announce "[Segurança do Calabouço] Parabéns aos " + getmapusers("job3_rune02") + " venderores!!", bc_all, 0xADD8E6;
				monster "job3_rune02",39,40,"Baú do Tesouro",1798,getmapusers("job3_rune02"),"Treasure::OnPriceBoxBreak";
				break;
		}
		end;
	
	EventDone:
		stopnpctimer;
		mapwarp "job3_rune02","prontera",122,161;
		end;
	
	RoundDone:
		set .round, getarg(0) +1;
		mapannounce "job3_rune02","[Evento Mecânico Pistola] 1 caixa de ferramenta foi derrubada pelo Mecânico Pistola !",bc_map,0xADD8E6;
		monster "job3_rune02",39,40,"Baú do Tesouro",1798,1,"Treasure::OnBoxBreak";
		sleep2 1000;
		mapannounce "job3_rune02","[Evento Mecânico Pistola] O " + getarg(0) + " Round Começou !",bc_map,0xADD8E6;
		end;
}
